# 前回課題解説

## ITP1_5: パターン出力

### ITP1_5_A: Print a Rectangle

高さ H、幅 W の長方形を `#` で出力します。

```c
#include <stdio.h>

int main() {
    int h, w;
    scanf("%d %d", &h, &w);
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            printf("#");
        }
        printf("\n");
    }
    return 0;
}
```

#### C++版

```cpp

using namespace std;

int main() {
    int h, w;

        for (int j = 0; j < w; j++) {
            cout << '#';
        }
        cout << '\n';
    }
    return 0;
}
```

### ITP1_5_B: Print a Frame

外枠だけ `#`、内側は `.` で出力します。

```c
#include <stdio.h>

int main() {
    int h, w;
    scanf("%d %d", &h, &w);
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            if (i == 0 || i == h - 1 || j == 0 || j == w - 1) {
                printf("#");
            } else {
                printf(".");
            }
        }
        printf("\n");
    }
    return 0;
}
```

#### C++版

### ITP1_5_D: Structured Programming

関数に処理を分割してみましょう。

```c
#include <stdio.h>

void print_rectangle(int h, int w) {
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) printf("#");
        printf("\n");
    }
}

void print_frame(int h, int w) {
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            if (i == 0 || i == h-1 || j == 0 || j == w-1)
                printf("#");
            else
                printf(".");
        }
        printf("\n");
    }
}

void print_checker(int h, int w) {
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++)
            printf("%c", (i + j) % 2 == 0 ? '#' : '.');
        printf("\n");
    }
}

int main() {
    int h, w;
    scanf("%d %d", &h, &w);
    print_rectangle(h, w);
    printf("\n");
    print_frame(h, w);
    printf("\n");
    print_checker(h, w);
    return 0;
}
```

## ITP1_6: 配列操作

### ITP1_6_A: Reversing Numbers

```c
#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);
    int a[1000];
    for (int i = 0; i < n; i++) scanf("%d", &a[i]);
    for (int i = n - 1; i >= 0; i--) {
        if (i > 0) printf("%d ", a[i]);
        else       printf("%d\n", a[i]);
    }
    return 0;
}
```

#### C++版

```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> a(n);
    for (int i = 0; i < n; i++) cin >> a[i];
    for (int i = n - 1; i >= 0; i--) {
        if (i > 0) cout << a[i] << ' ';
        else       cout << a[i] << '\n';
    }
    return 0;
}
```

### ITP1_6_B: Finding Missing Cards

52 枚のトランプから抜けているカードを探します。

```c
#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);
    int has[5][14] = {0};  // スート×ランク
    for (int i = 0; i < n; i++) {
        char s[4];
        int r;
        scanf("%s %d", s, &r);
        int suit;
        if (s[0] == 'S') suit = 1;
        else if (s[0] == 'H') suit = 2;
        else if (s[0] == 'C') suit = 3;
        else suit = 4;
        has[suit][r] = 1;
    }
    for (int suit = 1; suit <= 4; suit++) {
        char name[] = {'S', 'H', 'C', 'D'};
        for (int r = 1; r <= 13; r++) {
            if (!has[suit][r]) {
                printf("%c %d\n", name[suit - 1], r);
            }
        }
    }
    return 0;
}
```

#### C++版

```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<vector<int>> has(5, vector<int>(14, 0));

    for (int i = 0; i < n; i++) {
        char s;
        int r;
        cin >> s >> r;

        int suit;
        if (s == 'S') suit = 1;
        else if (s == 'H') suit = 2;
        else if (s == 'C') suit = 3;
        else suit = 4;

        has[suit][r] = 1;
    }

    char name[] = {'S', 'H', 'C', 'D'};
    for (int suit = 1; suit <= 4; suit++) {
        for (int r = 1; r <= 13; r++) {
            if (!has[suit][r]) {
                cout << name[suit - 1] << ' ' << r << '\n';
            }
        }
    }
    return 0;
}
```

:::tip ポイント
2次元配列 `has[suit][rank]` を使ってフラグ管理するのが典型パターンです。
:::
