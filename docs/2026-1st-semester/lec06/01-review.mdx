## ALDS1_1_A: Insertion Sort

挿入ソートを実行し、各ステップ終了時の配列を空白区切りで出力します。

```c
#include <stdio.h>

int main(void) {
    int n;
    scanf("%d", &n);

    int a[100];
    for (int i = 0; i < n; i++) scanf("%d", &a[i]);

    for (int i = 0; i < n; i++) {
        if (i > 0) printf(" ");
        printf("%d", a[i]);
    }
    printf("\n");

    for (int i = 1; i < n; i++) {
        int v = a[i];
        int j = i - 1;
        while (j >= 0 && a[j] > v) {
            a[j + 1] = a[j];
            j--;
        }
        a[j + 1] = v;

        for (int k = 0; k < n; k++) {
            if (k > 0) printf(" ");
            printf("%d", a[k]);
        }
        printf("\n");
    }

    return 0;
}
```

### C++版

```cpp
#include <iostream>
#include <vector>
using namespace std;

void print_array(const vector<int>& a) {
    for (int i = 0; i < (int)a.size(); i++) {
        if (i > 0) cout << ' ';
        cout << a[i];
    }
    cout << '\n';
}

int main() {
    int n;
    cin >> n;

    vector<int> a(n);
    for (int i = 0; i < n; i++) cin >> a[i];

    print_array(a);

    for (int i = 1; i < n; i++) {
        int v = a[i];
        int j = i - 1;
        while (j >= 0 && a[j] > v) {
            a[j + 1] = a[j];
            j--;
        }
        a[j + 1] = v;
        print_array(a);
    }

    return 0;
}
```

:::tip ポイント
最初に「初期配列」を出力してから、`i = 1` 以降の各ステップごとに配列全体を出力します。
出力形式の取り違えは AOJ で WA の代表的な原因です。
:::

## ALDS1_1_B: Greatest Common Divisor

ユークリッドの互除法で `gcd(x, y)` を求めます。

```c
#include <stdio.h>

int gcd(int a, int b) {
    while (b != 0) {
        int r = a % b;
        a = b;
        b = r;
    }
    return a;
}

int main(void) {
    int x, y;
    scanf("%d %d", &x, &y);
    printf("%d\n", gcd(x, y));
    return 0;
}
```

### C++版

```cpp

using namespace std;

int gcd(int a, int b) {
    while (b != 0) {
        int r = a % b;
        a = b;
        b = r;
    }
    return a;
}

int main() {
    int x, y;
    cin >> x >> y;
    cout << gcd(x, y) << '\n';
    return 0;
}
```

## ALDS1_1_C: Prime Numbers

`N` 個の整数のうち、素数の個数を数えます。

```c
#include <stdio.h>

int is_prime(int n) {
    if (n < 2) return 0;
    if (n == 2) return 1;
    if (n % 2 == 0) return 0;
    for (int i = 3; (long long)i * i <= n; i += 2) {
        if (n % i == 0) return 0;
    }
    return 1;
}

int main(void) {
    int n;
    scanf("%d", &n);

    int cnt = 0;
    for (int i = 0; i < n; i++) {
        int x;
        scanf("%d", &x);
        if (is_prime(x)) cnt++;
    }
    printf("%d\n", cnt);
    return 0;
}
```

### C++版

```cpp
#include <iostream>
using namespace std;

bool is_prime(int n) {
    if (n < 2) return false;
    if (n == 2) return true;
    if (n % 2 == 0) return false;
    for (long long i = 3; i * i <= n; i += 2) {
        if (n % i == 0) return false;
    }
    return true;
}

int main() {
    int n;
    cin >> n;

    int cnt = 0;
    for (int i = 0; i < n; i++) {
        int x;
        cin >> x;
        if (is_prime(x)) cnt++;
    }
    cout << cnt << '\n';
    return 0;
}
```

:::warning オーバーフローに注意
`i * i` の比較は `int` の範囲を超える可能性があるため、`long long` にキャストしておくと安全です。
:::

## ALDS1_1_D: Maximum Profit

時刻順に並ぶ価格列 `R[0], R[1], ..., R[N-1]` に対して、`R[j] - R[i]`（`i < j`）の最大値を求めます。

```c
#include <stdio.h>

int main(void) {
    int n;
    scanf("%d", &n);

    int r;
    scanf("%d", &r);

    int minv = r;
    int maxd = -2000000000;  // 十分小さい初期値

    for (int i = 1; i < n; i++) {
        scanf("%d", &r);
        if (r - minv > maxd) maxd = r - minv;
        if (r < minv) minv = r;
    }

    printf("%d\n", maxd);
    return 0;
}
```

### C++版

```cpp
#include <iostream>
#include <climits>
using namespace std;

int main() {
    int n;
    cin >> n;

    int r;
    cin >> r;

    int minv = r;
    int maxd = INT_MIN;

    for (int i = 1; i < n; i++) {
        cin >> r;
        if (r - minv > maxd) maxd = r - minv;
        if (r < minv) minv = r;
    }

    cout << maxd << '\n';
    return 0;
}
```

:::tip 更新順序がカギ
**先に最大利益を更新してから、最小値を更新する**のがポイントです。
逆にしてしまうと、同じ時刻の値で売買することになり、`i < j` という条件を満たさなくなります。
:::
