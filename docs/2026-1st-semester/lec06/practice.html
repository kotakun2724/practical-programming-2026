練習問題

ここまで学んだ計算量・ソート・探索を、実際に手を動かして確かめましょう。

練習 1: 最大値と最小値を1回の走査で求める​

n 個の整数を読み込み、最大値と最小値を出力するプログラムを書いてください。 ループは1回だけ（O(N)）に抑えるのが目標です。

#include <stdio.h>

int main(void) {
    int n;
    scanf("%d", &n);

    int x;
    scanf("%d", &x);
    int minv = x, maxv = x;

    for (int i = 1; i < n; i++) {
        scanf("%d", &x);
        // ここを実装
    }

    printf("%d %d\n", minv, maxv);
    return 0;
}
解答例
for (int i = 1; i < n; i++) {
    scanf("%d", &x);
    if (x < minv) minv = x;
    if (x > maxv) maxv = x;
}

練習 2: バブルソートを関数として実装する​

昇順にソートする bubble_sort 関数を完成させてください。

#include <stdio.h>

void bubble_sort(int *a, int n) {
    // ここを実装
}

int main(void) {
    int a[] = {5, 2, 4, 6, 1, 3};
    int n = 6;
    bubble_sort(a, n);
    for (int i = 0; i < n; i++) printf("%d ", a[i]);
    printf("\n");  // 1 2 3 4 5 6
    return 0;
}
解答例
void bubble_sort(int *a, int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = n - 1; j > i; j--) {
            if (a[j - 1] > a[j]) {
                int t = a[j - 1];
                a[j - 1] = a[j];
                a[j] = t;
            }
        }
    }
}

練習 3: 二分探索関数の実装​

ソート済み配列 a から key を二分探索する関数を実装してください。 見つかったらそのインデックス、見つからなければ -1 を返します。

int bsearch_idx(int *a, int n, int key) {
    // ここを実装
    return -1;
}
解答例
int bsearch_idx(int *a, int n, int key) {
    int lo = 0, hi = n - 1;
    while (lo <= hi) {
        int mid = (lo + hi) / 2;
        if (a[mid] == key) return mid;
        else if (a[mid] < key) lo = mid + 1;
        else hi = mid - 1;
    }
    return -1;
}

AOJ ALDS1_2_A: Bubble Sort​

n 個の整数を昇順にバブルソートして出力し、最後に交換回数を出力します。

入力形式​

n
a_0 a_1 ... a_{n-1}

出力形式​

1行目に空白区切りのソート後の配列、2行目に交換回数を出力します。

ヒント​

#include <stdio.h>

int main(void) {
    int n;
    scanf("%d", &n);

    int a[100];
    for (int i = 0; i < n; i++) scanf("%d", &a[i]);

    int swap_count = 0;
    for (int i = 0; i < n - 1; i++) {
        for (int j = n - 1; j > i; j--) {
            if (a[j - 1] > a[j]) {
                int t = a[j - 1];
                a[j - 1] = a[j];
                a[j] = t;
                swap_count++;
            }
        }
    }

    for (int i = 0; i < n; i++) {
        if (i > 0) printf(" ");
        printf("%d", a[i]);
    }
    printf("\n%d\n", swap_count);
    return 0;
}

:::tip 内側ループの向きに注意 j = n - 1; j > i; j-- のように後ろから前へ走査するのが ALDS1_2_A の典型実装です。 前から走査するパターンもありますが、AOJ の想定解と比較するなら後ろからが分かりやすいです。 :::

AOJ ALDS1_4_B: Binary Search​

ソート済みの配列 S の中に、別の配列 T の各値がいくつ含まれているかを数えます。

ヒント​

二分探索を使うと、各 T[i] について O(log N) で判定でき、全体で O(M log N) になります。 線形探索（O(NM)）では TLE する可能性があるので注意しましょう。

int count = 0;
for (int i = 0; i < m; i++) {
    int key;
    scanf("%d", &key);
    int lo = 0, hi = n - 1;
    while (lo <= hi) {
        int mid = (lo + hi) / 2;
        if (s[mid] == key) { count++; break; }
        else if (s[mid] < key) lo = mid + 1;
        else hi = mid - 1;
    }
}
printf("%d\n", count);

:::note ソート済みの前提を忘れない S がソート済みであることが二分探索の前提です。 入力がソート済みでない場合は、先に並べ替える必要があります。 :::
